Import Session from Browser

                
            

            

                

                    
1. Open Chrome DevTools Console on the logged-in site

                    
2. Paste this script and press Enter:

                    
copy(JSON.stringify({
  _tetraSession: true,
  url: location.href,
  origin: location.origin,
  cookies: document.cookie.split(';').map(c => {
    const [n,...v] = c.trim().split('=');
    return {name: n, value: v.join('=')};
  }).filter(c => c.name),
  localStorage: Object.keys(localStorage).map(k => ({name: k, value: localStorage[k]})),
  jwt: localStorage.getItem('token') || localStorage.getItem('accessToken') ||
       localStorage.getItem('jwt') || null
}, null, 2));
